# 回溯

```js
const res = [];
const backtrack = (开始位置, 当前选择列表) => {
  if (满足条件) {
    res.push([...当前选择列表]);
    return;
  }

  for (let i = 开始位置; i < 总数据集的尾巴; i++) {
    // 选择列表填充新的值
    当前选择列表.push(开始位置对应的值);
    // 继续向下，这里需要注意
    backtrack(i, 根据场景组装下一轮的选择列表);
    // 选择列表还原
    当前选择列表.pop();
  }
};
```
